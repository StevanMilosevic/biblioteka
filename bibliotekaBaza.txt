CREATE DATABASE biblioteka

CREATE TABLE biblioteke(
	bibl_id int PRIMARY KEY AUTO_INCREMENT,
    naziv varchar(40),
    ulica varchar(40),
    broj int(3),
    telefon varchar(40)
) ENGINE=InnoDB

CREATE TABLE clanovi(
	clan_karta_id int PRIMARY KEY AUTO_INCREMENT,
    ime varchar(20),
    prezime varchar(40),
    dat_rodj DATE,
    ulica varchar(35),
    broj varchar(10),
    telefon varchar(40)
) ENGINE=INNODB

CREATE TABLE knjige(
    knjiga_id int PRIMARY KEY AUTO_INCREMENT,
    naziv varchar(20),
    jezik varchar(40),
    izdavac varchar(35),
    bibl_id int,
    poz_od_id int
) ENGINE=INNODB


CREATE TABLE pozajmno_odeljenje(
    poz_od_id int PRIMARY KEY AUTO_INCREMENT,
    aktuelnost varchar(30),
    zanr varchar(15),
    decije_id int,
    naucno_id int
) ENGINE=INNODB

CREATE TABLE decije_knjige(
    decije_id int PRIMARY KEY AUTO_INCREMENT,
    uzrast varchar(15)
) ENGINE=INNODB

CREATE TABLE naucne_knjige(
    naucno_id int PRIMARY KEY AUTO_INCREMENT,
    oblast varchar(30)
) ENGINE=INNODB

CREATE TABLE autori(
    autor_id int PRIMARY KEY AUTO_INCREMENT,
    ime varchar(30),
    prezime varchar(45)
) ENGINE=INNODB

CREATE TABLE autori_knjige(
    autori_knjige_id int PRIMARY KEY AUTO_INCREMENT,
    autor_id int,
    knjiga_id int
) ENGINE=INNODB

CREATE TABLE bibliotekari(
    bibliotekar_id int PRIMARY KEY AUTO_INCREMENT,
    ime varchar(30),
    prezime varchar(45),
    ulica varchar(45),
    broj varchar(10),
    jmbg int(13),
    telefon varchar(20),
    bibl_id int
) ENGINE=INNODB

CREATE TABLE uzete_knjige(
    uzeto_id int PRIMARY KEY AUTO_INCREMENT,
    clan_karta_id int,
    bibliotekar_id int,
    knjiga_id int,
    datum_izdavanja DATE,
    datum_vracanja DATE    
) ENGINE=INNODB

CREATE TABLE bibl_clanovi(
    bibl_clan_id int PRIMARY KEY AUTO_INCREMENT,
    bibl_id int,
   clan_karta_id int   
) ENGINE=INNODB
